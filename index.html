const music = document.getElementById('bgMusic');

// Countdown Timer (example: change date to your wedding date)

const countdown = document.getElementById('countdown');

const weddingDate = new Date("2025-12-31T17:00:00"); // update to your wedding date

function updateCountdown() {

  const now = new Date();

  const diff = weddingDate - now;

  if (diff <= 0) {

    countdown.innerHTML = "🎉 Մենք արդեն ամուսնացել ենք! 🎉";

  } else {

    const days = Math.floor(diff / (1000 * 60 * 60 * 24));

    const hours = Math.floor((diff / (1000 * 60 * 60)) % 24);

    const minutes = Math.floor((diff / (1000 * 60)) % 60);

    const seconds = Math.floor((diff / 1000) % 60);

    countdown.innerHTML = `Հարսանիքի մնացած ժամանակը: ${days} օր ${hours} ժամ ${minutes} րոպե ${seconds} վայրկյան`;

  }

}

setInterval(updateCountdown, 1000);

updateCountdown();

</script>
